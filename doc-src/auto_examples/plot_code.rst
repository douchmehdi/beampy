    from beampy import *

    # Remove quiet=True to get beampy compilation outputs

    with slide('A small python code'):
        c = code("""
        from pylab import *

        n = range(10)

        plot(n, n*random.rand(), 'ko')

        for i in range(10):
            print(i)

        """, width=500)

    display_matplotlib(gcs())
